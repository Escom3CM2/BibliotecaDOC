\begin{UseCase}{CU5.3}{Reponer credencial}{
	El lector podrá reponer su credencial para hacer uso de los servicios de la biblioteca.
}
	\UCitem{Versión}{1.1}
	\UCitem{Actor}{Lector.}
	\UCitem{Propósito}{Reposición de la credencial del lector.}
	\UCitem{Entradas}{ID de lector.}
	\UCitem{Salidas}{Se genera la credencial en formato PDF con la información correspondiente además del mensaje \MSGref{MSG5.1}{Credencial Lista}, \MSGref{MSG5.2}{Desea renovar credencial?.} o \MSGref{MSG5.3}{Error al conectar con la Base de Datos}  }
	\UCitem{Precondiciones}{	
	\begin{itemize}
	\item El lector debe ser estudiante o docente.

  \end{itemize}
}

	\UCitem{Postcondiciones}{
	\begin{itemize}
		   \item El lector podra hacer uso de los servicios de la biblioteca.
		  \end{itemize}
		}
	\UCitem{Autores}{
	\begin{itemize}
		\item Manriquez Molina Diego
		\item Mora Daniel
		\end{itemize}
		}

	
\end{UseCase}


%%trayectoria principal

\begin{UCtrayectoria}{Principal}
	\UCpaso[\UCactor] Selecciona el botón  de \IUbutton{Reponer credencial}en la pantalla\IUref{IU5.02}{ReposicionCredencial}.
	\UCpaso[\UCsist] Despliega el \MSGref{MSG5.2}{Desea renovar credencial?.}
	\UCpaso[\UCactor] Da clic en el botón  \IUbutton{Renovar}.
	\UCpaso[\UCsist] Busca la información en la base de datos. \Trayref{A}
	\UCpaso[\UCsist] Despliega el \MSGref{MSG5.1}{Credencial Lista}
	\UCpaso[\UCactor]Da clic en el botón \IUbutton{Continuar}.
	\UCpaso[\UCsist] Despliega la pantalla \IUref{IU5.01} creacionCredencial.
	\UCpaso[\UCsist] Muestra el PDF generado de la credencial.
	
\end{UCtrayectoria}

\begin{UCtrayectoriaA}{A}{Fallo de conexión en la base de datos}
			\UCpaso[\UCsist] El sistema mostrará el mensaje \MSGref{MSG5.3}{Error al conectar con la Base de Datos}.
			\UCpaso[\UCsist] Regresa a la pantalla \IUref{IU5.00}{RegStudent}	
\end{UCtrayectoriaA}
